\begin{DoxyItemize}
\item un header \+: une suite de Connection\+::mcs\+\_\+header\+Size uint32\+\_\+t (donc taille fixe)

\end{DoxyItemize}

\end{DoxyItemize}

\end{DoxyItemize}
